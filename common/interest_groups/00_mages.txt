ig_mages = {
	color = hsv{ 0.81 0.56 0.75 }
	texture = "gfx/interface/icons/ig_icons/mages.dds"
	layer = "revolution_dynamic_mages"
	index = 8

	ideologies = {
		ideology_arcane_scholars
		ideology_power_hungry
	}

	enable = {
		always = yes
	}

	# Deprecated; use on_enable effect to assign traits instead
	traits = {
		ig_trait_solidarity
		ig_trait_industrial_organizers
		ig_trait_work_to_rule
	}
	
	on_enable = {
		ig:ig_mages = {
			set_ig_trait = ig_trait:ig_trait_work_to_rule
			set_ig_trait = ig_trait:ig_trait_industrial_organizers
			set_ig_trait = ig_trait:ig_trait_solidarity
		}

		if = {
			limit = {
				owner = { 
					any_primary_culture = {
						has_discrimination_trait = dwarf_heritage
					}	
				}
			}
			ig:ig_mages = {
				set_interest_group_name = ig_runeseers
			}
		}

	}

	on_disable = {}
	on_character_ig_membership = {}

	pop_potential = {
		literacy_rate > 0.25
		OR = {
			is_pop_type = aristocrats
			is_pop_type = bureaucrats
			is_pop_type = capitalists
			is_pop_type = engineers
			is_pop_type = wizards
			is_pop_type = clergymen
			AND = {
				is_pop_type = academics
				owner = {
					has_law = law_type:law_astrological_studies
				}
			}
		}
	}

	pop_weight = {
		value = 0

		add = {
			desc = "POP_WIZARDS"
			if = {
				limit = {
					is_pop_type = wizards
				}
				value = 250
			}
		}

		add = {
			desc = "POP_BUREAUCRATS"
			if = {
				limit = {
					is_pop_type = bureaucrats
				}
				value = 100
			}
		}

		if = {
			limit = {
				owner = {
					has_law = law_type:law_court_mages
				}
			}
			add = {
				desc = "POP_BUREAUCRATS_MAGES"
				if = {
					limit = {
						is_pop_type = bureaucrats
					}
					value = 100
				}
			}
		}

		add = {
			desc = "ASTROLOGICAL_STUDIES_LAW"
			if = {
				limit = {
					is_pop_type = wizards
					owner = {
						has_law = law_type:law_astrological_studies
					}
				}
				value = 25
			}
		}

		if = {
			limit = {
				owner = {
					has_law = law_type:law_court_mages
				}
			}
			add = {
				desc = "POP_BUREAUCRATS_MAGES"
				if = {
					limit = {
						is_pop_type = bureaucrats
					}
					value = 50
				}
			}
		}

		add = {
			desc = "POP_ARISTOCRATS"
			if = {
				limit = {
					is_pop_type = aristocrats
				}
				if = {
					limit = {
						owner = {
							has_law = law_type:law_hereditary_bureaucrats
						}
					}
					value = 75
				}
				else = {
					value = 25
				}
			}
		}

		add = {
			desc = "ADMINISTRATION_ROLE"
			if = {
				limit = {
					OR = {
						is_pop_type = aristocrats
						is_pop_type = clergymen
						is_pop_type = capitalists
						is_pop_type = wizards
					}
					OR = {
						pop_employment_building_group = bg_service
						pop_employment_building_group = bg_arts
						pop_employment_building_group = bg_government
					}
				}
				value = 100
			}
		}
		
		if = {
			limit = {
				literacy_rate >= 0.50
			}
			add = {
				desc = "POP_LITERACY"
				value = literacy_rate
				multiply = 20
			}
		}

		multiply = {
			desc = "LEADER_POPULARITY"
			scope:interest_group = {
				leader ?= {
					value = popularity
					multiply = 0.0025
					add = 1
					max = 1.25
					min = 0.75
				}
			}
		}

		multiply = {
			desc = "decentralized_power"
			value = 1.0

			if = {
				limit = {
					owner = {
						is_country_type = decentralized
					}
				}
				value = 0.0
			}
		}
		if = {
			limit = {
				scope:interest_group = {
					is_in_government = yes
				}
			}
			multiply = {
				desc = "IN_GOVERNMENT_ATTRACTION"
				value = 1
				add = scope:interest_group.modifier:interest_group_in_government_attraction_mult
				min = 0
			}
		}
		
		if = {
			limit = {
				scope:interest_group = {
					is_in_government = yes
				}
			}
			multiply = { # Multiplied by 1.25x max
				desc = "POP_LOYALISTS"
				value = pop_loyalist_fraction
				divide = 4
				add = 1
			}
		}
		if = {
			limit = {
				scope:interest_group = {
					is_in_government = no
				}
			}
			multiply = { # Multiplied by 1.25x max
				desc = "POP_RADICALS"
				value = pop_radical_fraction
				divide = 4
				add = 1
			}
		}
	}

	monarch_weight = {
		value = 1
		# Monarch is not likely to adopt a marginal IG
		if = {
			limit = {
				is_marginal = yes
			}
			multiply = {
				value = 0.1
			}
		}
		# Monarch is more likely to adopt a Powerful IG
		if = {
			limit = {
				is_powerful = yes
			}
			multiply = {
				value = 2
			}
		}
		# Much more likely with Technocracy
		if = {
			limit = {
				owner = {
					has_law = law_type:law_technocracy
				}
			}
			multiply = {
				value = 3
			}
		}
		# More likely with Court Mages
		if = {
			limit = {
				owner = {
					has_law = law_type:law_court_mages
				}
			}
			multiply = {
				value = 1.5
			}
		}
		# Impossible without Empiricism
		if = {
			limit = {
				owner = {
					NOT = { has_technology_researched = empiricism }
				}
			}
			multiply = {
				value = 0
			}
		}
	}

	agitator_weight = {
		# Agitators more likely to come from populist IG's rather than elitist
		value = 0.75
		# Agitators will never be part of a marginalized IG
		if = {
			limit = {
				owner.ig:ig_intelligentsia = {
					is_marginal = yes
				}
			}
			multiply = {
				value = 0
			}
		}
	}

	commander_weight = {
		value = 1.0
	}

	noble_chance = {
		value = 0.5
	}

	female_commander_chance = {
		value = 0.0

		if = {
			limit = {
				owner = {
					has_law = law_type:law_matriarchy
				}

			}
			add = {
				value = 1.0
			}
		}
	}

	female_politician_chance = {
		value = 0.0

		if = {
			limit = {
				owner = {
					has_law = law_type:law_matriarchy
				}

			}
			add = {
				value = 0.965
			}
		}
		else_if = {
			limit = {
				owner = {
					has_law = law_type:law_womens_suffrage
				}

			}
			add = {
				value = 0.035
			}
		}
	}

	female_agitator_chance = {
		value = 0.035
		
		if = {
			limit = {
				owner = {
					has_law = law_type:law_women_own_property
				}

			}
			add = {
				value = 0.035
			}
		}	
		
		if = {
			limit = {
				owner = {
					has_law = law_type:law_women_in_the_workplace
				}

			}
			add = {
				value = 0.035
			}
		}

		if = {
			limit = {
				owner = {
					has_law = law_type:law_womens_suffrage
				}

			}
			add = {
				value = 0.035
			}
		}		

		if = {
			limit = {
				owner = {
					has_law = law_type:law_matriarchy
				}

			}
			add = {
				value = 0.95
			}
		}
	}

	# The chance that a commander belonging to this IG takes over leadership when it changes
	# scope:character is the most popular commander in the IG
	commander_leader_chance = {
		value = 0.25
		if = {
			limit = {
				exists = scope:character
			}
			multiply = {
				value = 1
				add = scope:character.popularity
				min = 0
			}
		}
	}
}
